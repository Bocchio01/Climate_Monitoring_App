package Finestra;

import java.awt.Container;
import javax.swing.ImageIcon;
import javax.swing.JFrame;

//Classe astratta per i singoli frame
public abstract class SchermataBase extends JFrame {

    protected Container container = getContentPane();
    protected ImageIcon icona;

    public SchermataBase(String titolo) {
        super(titolo);
        setDefaultProperties();
        setLayoutManager();
        setLocationAndSize();
        addComponentsToContainer();
        addActionEvent();
    }

    protected void setDefaultProperties() {
        setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
        setResizable(false);
        icona = new ImageIcon("Immagini/logo_png.png");
        setIconImage(icona.getImage());
        setLocationRelativeTo(null);
    }

    protected abstract void setLayoutManager();

    protected abstract void setLocationAndSize();

    protected abstract void addComponentsToContainer();

    protected abstract void addActionEvent();
}
